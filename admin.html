Configurações do Banco de Dados (Supabase)

Configure a conexão direta com o Supabase para que a página grave os leads e carregue as lojas de forma permanente na nuvem.

Encontrado em: *Project Settings > API > Project URL* no painel do Supabase.

Encontrado em: *Project Settings > API > Project API keys (anon public)*.

Tabela no banco de dados que irá receber os campos: `name`, `email`, `whatsapp`, `message`, `loja`, `voucher`.

Tabela no banco de dados que irá receber as óticas: `nome`, `endereco`, `telefone`.

Opcional. Se configurado, a página enviará os dados do lead diretamente do navegador para o Make, sem depender de gatilhos do Supabase.

Não configurado / Sem conexão local

🛠️ Script SQL para criação das tabelas no Supabase:

Copie e execute o código abaixo no menu **SQL Editor** do seu painel do Supabase para criar ou atualizar as tabelas do projeto:

-- 1. Tabela de Leads Promocionais com Loja e Voucher
create table if not exists leads_personality (
  id uuid default gen_random_uuid() primary key,
  name text not null,
  email text not null,
  whatsapp text not null,
  message text,
  loja text,
  loja_endereco text,
  loja_telefone text,
  voucher text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. Tabela de Lojas Licenciadas
create table if not exists lojas_licenciadas (
  id uuid default gen_random_uuid() primary key,
  nome text not null,
  endereco text,
  telefone text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 3. Tabela de Acessos de Lojas Credenciadas
create table if not exists acessos_lojas (
  id uuid default gen_random_uuid() primary key,
  loja_nome text not null unique,
  usuario text not null unique,
  senha text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 4. Tabela de Equipe Técnica (Suporte & Garantia)
create table if not exists tecnicos_personality (
  id uuid default gen_random_uuid() primary key,
  nome text not null,
  email text not null,
  whatsapp text not null,
  especialidade text default 'Assistência Geral',
  loja_atendida text default 'Todas as Lojas',
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 5. Tabela de Chamados de Assistência Técnica (Com O.S., CRM Médico, Tratamento e Data da Compra)
create table if not exists chamados_assistencia (
  id uuid default gen_random_uuid() primary key,
  protocolo text not null,
  cliente_nome text not null,
  cliente_email text not null,
  cliente_whatsapp text not null,
  loja_nome text not null,
  linha_produto text not null,
  tratamento text not null,
  data_compra date not null,
  os_numero text not null,
  medico_crm text not null,
  motivo text not null,
  data_atendimento date not null,
  horario_atendimento text not null,
  tecnico_nome text not null,
  tecnico_email text not null,
  tecnico_whatsapp text not null,
  status text default 'Agendado',
  observacoes text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 6. Tabela de Motivos do Chamado
create table if not exists motivos_assistencia (
  id uuid default gen_random_uuid() primary key,
  motivo text not null unique,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 7. Liberar acesso às tabelas (desativar RLS para API pública)
alter table leads_personality disable row level security;
alter table lojas_licenciadas disable row level security;
alter table acessos_lojas disable row level security;
alter table tecnicos_personality disable row level security;
alter table chamados_assistencia disable row level security;
alter table motivos_assistencia disable row level security;
